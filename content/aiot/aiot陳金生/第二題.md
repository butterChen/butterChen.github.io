module v2(clk,c,r,seg);
	input clk;
	input [2:0]c;
	output reg [3:0]r;
	output reg [6:0]seg;

	always @ (posedge clk) begin
	r=r<<1;
	if(r==0) r=1;
	
	case(r)
2:begin
if(c==1)seg<=79;
if(c==2)seg<=18;
if(c==4)seg<=6;
end
4:begin
if(c==1)seg<=76;
if(c==2)seg<=36;
if(c==4)seg<=32;
end
8:begin
if(c==1)seg<=13;
if(c==2)seg<=0;
if(c==4)seg<=4;
end
1:begin
if(c==1)seg<=48;
if(c==2)seg<=1;
if(c==4)seg<=72;
end
endcase
end
endmodule
