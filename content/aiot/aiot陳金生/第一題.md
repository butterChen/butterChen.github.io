module v1(clk,sc,seg);

	input clk;
	output reg [3:0] sc;
	output reg [7:0] seg;
	reg[11:0]f;
	
	always @ (posedge clk )
	f=f+1;
	always @ (posedge f[11] )
	begin
		sc=sc<<1;
		if (sc == 0) sc=1;
		case (sc)
	            1: seg <=8'b00110000;
	            2: seg <=8'b11111110;
	            4: seg <=8'b01111111;
	            8: seg <=8'b01111110;
	endcase
	end

endmodule
